Aula 09 - Comandos de Seleção

if-else

O comando if é o mais básico de todos os comandos de seleção e deve ser usado com as duas letras minúsculas. Ele determina que o programa deve executar um certo comando ou seção de código apenas se uma certa condição for verdadeira.

Comando de Seleção if

Exemplo 1 (script):
function situacao(entrada1, entrada2, saida) {
var x = Number(document.getElementById(entrada1).value);
var y = Number(document.getElementById(entrada2).value);
var media = (x + y)/2;

document.getElementById(saida).innerHTML = "O resultado é " + resultado(media);
}

function resultado(m) {
var resultado;
if (m >= 7) {
resultado = resultado + ". Parabéns!!!";
}
if (m < 7){
resultado = "Reprovado";
resultado = resultado;
}
return resultado;
}

Comando de Seleção if-else Encadeado

É importante ressaltar que não existe um limite para o encadeamento de condições.

Exemplo 2(script):
function situacao2(entrada1, entrada2, saida) {
var x = Number(document.getElementById(entrada1).value);
var y = Number(document.getElementById(entrada2).value);
var media = (x + y)/2;
document.getElementById(saida2).innerHTML = "O resultado é " + resultado(media);
}

function resultado(m) {
var resultado;
if (m >= 7) {
resultado = "Aprovado";
} else if (m >=5) {
resultado = "Recuperação";
} else {
resultado = "Reprovado";
}
return resultado;
}
